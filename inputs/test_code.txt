static void phex(uint8_t* str)
{

    uint8_t len = 16;
    unsigned char i;
    for (i = 0; i < len; ++i)
        printf("%.2x", str[i]);
    printf("\n");
}

uint8_t rkey[176];
    for (int i = 0; i < 176; i++) {
        rkey[i] = i;

    AddRoundKey(0, &state, rkey);
